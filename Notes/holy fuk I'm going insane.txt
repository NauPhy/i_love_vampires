class Stat{
	float _final;
public:
	const float _base;
	float _prebonus;
	float _postbonus;
	float _multiplier;
	float _offset;
	stat() = delete;
	stat(float base) : _base(base), _prebonus(0), _postbonus(0), _multiplier(0), _offset(0) {}
	float calculateFinal() {
		_final = (_base+_prebonus)*(1+_multiplier)+_postbonus+_offset;
		return _final;
	}
	float getFinal(){
		return _final;
	}
	void softReset(){
		_prebonus = 0;
		_postbonus = 0;
		_multiplier = 0;
	}
};
